-- View students with their results
SELECT 
    s.name AS student_name,
    s.roll_no,
    sub.subject_name,
    r.marks_obtained,
    sub.total_marks,
    r.grade,
    r.exam_type
FROM results r
JOIN students s ON r.student_id = s.id
JOIN subjects sub ON r.subject_id = sub.id;

-- View subjects with assigned teachers
SELECT 
    sub.subject_code,
    sub.subject_name,
    sub.class_name,
    t.name AS teacher_name,
    t.teacher_id
FROM subjects sub
LEFT JOIN teachers t ON sub.teacher_id = t.id;

-- Check total students per class
SELECT class_name, section, COUNT(*) as student_count
FROM students
GROUP BY class_name, section;
